import pandas as pd
import matplotlib.pyplot as plt

# Leer archivo CSV
df = pd.read_csv('barri.csv')

# Agrupar por distrito y contar los barrios
barrios_por_distrito = df.groupby('Nom_Districte')['Nom_Barri'].nunique()

# Calcular porcentaje de cada distrito
porcentaje_por_distrito = (barrios_por_distrito / barrios_por_distrito.sum()) * 100

# Crear grfico de torta
plt.pie(porcentaje_por_distrito.values, labels=porcentaje_por_distrito.index, autopct='%1.1f%%')
plt.title('Participacin porcentual de cada distrito por numeros de barrios')
plt.show()

import pandas as pd
import matplotlib.pyplot as plt

# Cargar datos en una tabla
df = pd.read_excel('nuevobarri.xls')

# Agrupar datos por nombre y trimestre y sumar los precios
grouped_data = df.groupby(['Nom_Districte', 'Trimestre'])['Preu'].sum()

# Reorganizar datos en una tabla con nombre y trimestre como ndices y precios como valores
grupos = grouped_data.unstack()

# Generar grfico apilado
grupos.plot(kind='bar', stacked=True)

# Agregar ttulo y etiquetas a los ejes
plt.title('Precios totalizados por trimestre y distritos')
plt.xlabel('Trimestre')
plt.ylabel('Precios')

# Mostrar grfico
plt.show()
